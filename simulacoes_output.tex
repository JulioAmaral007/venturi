\subsection{Simulação: Caso 1: Modo Ideal - Água}

\subsubsection{Parâmetros de Entrada}

\begin{itemize}
    \item Diâmetro de entrada: $D_1 = 0.1$ m
    \item Diâmetro da garganta: $D_2 = 0.05$ m
    \item Comprimento da garganta: $L_{garganta} = 0.05$ m
    \item Densidade do fluido: $\rho = 1000$ kg/m³
    \item Densidade do fluido manométrico: $\rho_m = 13600$ kg/m³
    \item Vazão volumétrica: $Q = 0.01$ m³/s
    \item Fator de atrito: $f = 0.02$
    \item Coeficiente de descarga: $C_d = 1.0$
    \item Viscosidade dinâmica: $\mu = 0.001$ Pa·s
    \item Pressão na entrada: $P_1 = 101325$ Pa
    \item Modo de operação: \textbf{Ideal}
\end{itemize}

\subsubsection{Cálculos Intermediários}

\paragraph{Áreas das Seções:}
\begin{align*}
    A_1 &= \pi \left(\frac{D_1}{2}\right)^2 = \pi \left(\frac{0.1}{2}\right)^2 = 0.007854 \text{ m}^2 \\
    A_2 &= \pi \left(\frac{D_2}{2}\right)^2 = \pi \left(\frac{0.05}{2}\right)^2 = 0.001963 \text{ m}^2
\end{align*}

\paragraph{Velocidades:}
\begin{align*}
    v_1 &= \frac{Q}{A_1} = \frac{0.01}{0.007854} = 1.2732 \text{ m/s} \\
    v_2 &= \frac{Q}{A_2} = \frac{0.01}{0.001963} = 5.0930 \text{ m/s}
\end{align*}

\paragraph{Coeficiente de Perda na Entrada:}
$k_{entrada} = 0.0$ (modo Ideal)

\paragraph{Pressão na Garganta (antes das perdas):}
\begin{align*}
    P_2 &= P_1 - \frac{1}{2}\rho \left[v_2^2(1 + k_{entrada}) - v_1^2\right] \\
    P_2 &= 101325 - \frac{1}{2} \cdot 1000 \left[(5.0930)^2 \cdot (1 + 0.0) - (1.2732)^2\right] \\
    P_2 &= 89166.46 \text{ Pa}
\end{align*}

\paragraph{Perdas na Garganta:}
Modo ideal: sem perdas ($\Delta P_{garganta} = 0$)

\paragraph{Pressão Final na Garganta:}
$P_{2,fim} = P_2 = 89166.46$ Pa

\paragraph{Diferença de Pressão:}
\begin{align*}
    \Delta P &= P_1 - P_2 = 101325 - 89166.46 = 12158.54 \text{ Pa}
\end{align*}

\paragraph{Pressão na Saída:}
Modo ideal: $P_3 = P_1 = 101325.00$ Pa (recuperação total)

\paragraph{Perda de Carga Total:}
Modo ideal: $h_L = 0$ (sem perdas)

\paragraph{Desnível Manométrico:}
\begin{align*}
    \Delta h &= \frac{\Delta P}{(ho_m - \rho) g} = \frac{12158.54}{(13600 - 1000) \cdot 9.81} = 0.098365 \text{ m}
\end{align*}

\paragraph{Número de Reynolds:}
\begin{align*}
    Re &= \frac{\rho v_1 D_1}{\mu} = \frac{1000 \cdot 1.2732 \cdot 0.1}{0.001} = 127323.95
\end{align*}

\paragraph{Geometria do Venturi:}
\begin{align*}
    L_{entrada} &= 0.1899 \text{ m} \\
    L_{saida} &= 0.1899 \text{ m} \\
    L_{total} &= L_{entrada} + L_{garganta} + L_{saida} = 0.4298 \text{ m}
\end{align*}

\subsection{Simulação: Caso 2: Modo Realista - Água}

\begin{itemize}

    \item Diâmetro da garganta: $D_2 = 0.05$ m
    \item Comprimento da garganta: $L_{garganta} = 0.05$ m

    \item Fator de atrito: $f = 0.02$
    \item Coeficiente de descarga: $C_d = 0.98$
    \item Viscosidade dinâmica: $\mu = 0.001$ Pa·s
    \item Pressão na entrada: $P_1 = 101325$ Pa
    \item Modo de operação: \textbf{Realista}
\end{itemize}

\subsubsection{Cálculos Intermediários}

\paragraph{Áreas das Seções:}
\begin{align*}
    A_1 &= \pi \left(\frac{D_1}{2}\right)^2 = \pi \left(\frac{0.1}{2}\right)^2 = 0.007854 \text{ m}^2 \\
    A_2 &= \pi \left(\frac{D_2}{2}\right)^2 = \pi \left(\frac{0.05}{2}\right)^2 = 0.001963 \text{ m}^2
\end{align*}

\paragraph{Velocidades:}
\begin{align*}
    v_1 &= \frac{Q}{A_1} = \frac{0.01}{0.007854} = 1.2732 \text{ m/s} \\
    v_2 &= \frac{Q}{A_2} = \frac{0.01}{0.001963} = 5.0930 \text{ m/s}
\end{align*}

\paragraph{Coeficiente de Perda na Entrada:}
$k_{entrada} = 0.04$ (modo Realista)

\paragraph{Pressão na Garganta (antes das perdas):}
\begin{align*}
    P_2 &= P_1 - \frac{1}{2}\rho \left[v_2^2(1 + k_{entrada}) - v_1^2\right] \\
    P_2 &= 101325 - \frac{1}{2} \cdot 1000 \left[(5.0930)^2 \cdot (1 + 0.04) - (1.2732)^2\right] \\
    P_2 &= 88647.69 \text{ Pa}
\end{align*}

\paragraph{Perdas na Garganta:}
\begin{align*}
    h_{f,garganta} &= f \frac{L_{garganta}}{D_2} \frac{v_2^2}{2g} \\
    h_{f,garganta} &= 0.02 \cdot \frac{0.05}{0.05} \cdot \frac{(5.0930)^2}{2 \cdot 9.81} \\
    h_{f,garganta} &= 0.026441 \text{ m} \\
    \Delta P_{garganta} &= h_{f,garganta} \cdot \rho \cdot g = 0.026441 \cdot 1000 \cdot 9.81 = 259.38 \text{ Pa}
\end{align*}

\paragraph{Pressão Final na Garganta:}
\begin{align*}
    P_{2,fim} &= P_2 - \Delta P_{garganta} = 88647.69 - 259.38 = 88388.31 \text{ Pa}
\end{align*}

\paragraph{Diferença de Pressão:}
\begin{align*}
    \Delta P &= P_1 - P_2 = 101325 - 88647.69 = 12677.31 \text{ Pa}
\end{align*}

\paragraph{Cálculo da Pressão na Saída ($P_3$):}
\begin{align*}
    \Delta P_{recuperacao} &= \frac{1}{2}\rho(v_2^2 - v_1^2) = \frac{1}{2} \cdot 1000 \left[(5.0930)^2 - (1.2732)^2\right] = 12158.54 \text{ Pa} \\
    K_{difusor} &= 0.2940 \\
    \Delta P_{difusor} &= K_{difusor} \cdot \frac{1}{2}\rho v_2^2 = 0.2940 \cdot \frac{1}{2} \cdot 1000 \cdot (5.0930)^2 = 3812.92 \text{ Pa} \\
    P_3 &= P_{2,fim} + \Delta P_{recuperacao} - \Delta P_{difusor} \\
    P_3 &= 88388.31 + 12158.54 - 3812.92 = 96733.93 \text{ Pa}
\end{align*}

\paragraph{Perdas Totais:}
\begin{align*}
    \Delta P_{entrada} &= 518.76 \text{ Pa} \\
    \Delta P_{garganta} &= 259.38 \text{ Pa} \\
    \Delta P_{difusor} &= 3812.92 \text{ Pa} \\
    \Delta P_{total} &= 4591.07 \text{ Pa} \\
    h_L &= \frac{\Delta P_{total}}{\rho g} = \frac{4591.07}{1000 \cdot 9.81} = 0.467999 \text{ m}
\end{align*}

\paragraph{Desnível Manométrico:}
\begin{align*}
    \Delta h &= \frac{\Delta P}{(ho_m - \rho) g} = \frac{12677.31}{(13600 - 1000) \cdot 9.81} = 0.102562 \text{ m}
\end{align*}

\paragraph{Número de Reynolds:}
\begin{align*}
    Re &= \frac{\rho v_1 D_1}{\mu} = \frac{1000 \cdot 1.2732 \cdot 0.1}{0.001} = 127323.95
\end{align*}

\paragraph{Geometria do Venturi:}
\begin{align*}
    L_{entrada} &= 0.1899 \text{ m} \\
    L_{saida} &= 0.1899 \text{ m} \\
    L_{total} &= L_{entrada} + L_{garganta} + L_{saida} = 0.4298 \text{ m}
\end{align*}

    \item Viscosidade dinâmica: $\mu = 0.001$ Pa·s
    \item Pressão na entrada: $P_1 = 200000$ Pa
    \item Modo de operação: \textbf{Realista}
\end{itemize}

\subsubsection{Cálculos Intermediários}

\paragraph{Áreas das Seções:}
\begin{align*}
    A_1 &= \pi \left(\frac{D_1}{2}\right)^2 = \pi \left(\frac{0.15}{2}\right)^2 = 0.017671 \text{ m}^2 \\
    A_2 &= \pi \left(\frac{D_2}{2}\right)^2 = \pi \left(\frac{0.075}{2}\right)^2 = 0.004418 \text{ m}^2

\begin{align*}
    v_1 &= \frac{Q}{A_1} = \frac{0.025}{0.017671} = 1.4147 \text{ m/s} \\

    P_2 &= 184349.00 \text{ Pa}
\end{align*}

\begin{align*}
    h_{f,garganta} &= f \frac{L_{garganta}}{D_2} \frac{v_2^2}{2g} \\

\end{align*}

\begin{align*}

\end{align*}

\begin{align*}
    \Delta P_{recuperacao} &= \frac{1}{2}\rho(v_2^2 - v_1^2) = \frac{1}{2} \cdot 1000 \left[(5.6588)^2 - (1.4147)^2\right] = 15010.55 \text{ Pa} \\

    P_3 &= P_{2,fim} + \Delta P_{recuperacao} - \Delta P_{difusor} \\
    P_3 &= 184041.59 + 15010.55 - 4707.31 = 194344.83 \text{ Pa}

    \Delta P_{difusor} &= 4707.31 \text{ Pa} \\
    \Delta P_{total} &= 5655.17 \text{ Pa} \\
    h_L &= \frac{\Delta P_{total}}{\rho g} = \frac{5655.17}{1000 \cdot 9.81} = 0.576470 \text{ m}
\end{align*}

\paragraph{Desnível Manométrico:}
\begin{align*}
    \Delta h &= \frac{\Delta P}{(ho_m - \rho) g} = \frac{15651.00}{(13600 - 1000) \cdot 9.81} = 0.126620 \text{ m}

\subsection{Influência dos Parâmetros}

\subsubsection{Influência da Razão de Diâmetros}

A razão $\beta = D_2/D_1$ afeta significativamente:
\begin{itemize}
    \item A diferença de pressão $\Delta P$ (maior $\beta$ menor $\Delta P$)
    \item As perdas no difusor (maior razão de áreas, maiores perdas)
    \item A recuperação de pressão na saída
\end{itemize}

\subsubsection{Influência da Vazão}

A vazão $Q$ influencia:
\begin{itemize}
    \item As velocidades $v_1$ e $v_2$ (proporcional)
    \item A diferença de pressão $\Delta P$ (proporcional ao quadrado)
    \item As perdas de carga (proporcionais ao quadrado da velocidade)
    \item O número de Reynolds (proporcional)
\end{itemize}

\subsubsection{Influência do Fator de Atrito}

O fator de atrito $f$ afeta diretamente:
\begin{itemize}
    \item As perdas na garganta: $h_f = f \frac{L}{D} \frac{v^2}{2g}$
    \item A pressão final na garganta $P_{2,fim}$
    \item A perda de carga total $h_L$
\end{itemize}
